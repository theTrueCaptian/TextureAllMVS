Extraction Rules

Name := /[A-Z][a-z]+\s*[A-Z][a-z]+/g
     from Line #1
     within Page 1
Phone := /(\(\d+\)\d+-\d+|\d{3}[\s-]\d{3}[\s-]\d{4})/
     from Line
     after Heading #1
School := /,([A-Za-z\s]+University[A-Za-z\s]+),/g
     from Section
Year := /\d\d\d\d/
     within School
OR
Year := /\d\d\d\d[A-Za-z\s\.]*-[A-Za-z\s\.]*\d\d\d\d/
     within School
Education := Permute ( School, Year )
     from Section
     within Page 1
